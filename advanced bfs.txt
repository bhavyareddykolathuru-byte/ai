from collections import deque

# Input section
n = int(input("Enter the number of nodes: "))
v = input("Enter the node names separated by space: ").split()

g = {x: [] for x in v}

m = int(input("Enter the number of edges: "))

print("Enter the edges (node1 node2):")
for _ in range(m):
    a, b = input().split()
    g[a].append(b)
    g[b].append(a)

start = input("Enter the starting node: ")

# BFS Initialization
q = deque([start])
visited = set()
parent = {start: None}

step = 1

def get_path(node):
    path = []
    while node is not None:
        path.append(node)
        node = parent[node]
    return path[::-1]

# BFS Process
while q:
    x = q.popleft()
    visited.add(x)

    unvisited = set(v) - visited

    print(f"\nStep {step}")
    print("Source node :", x)
    print("Adjacent    :", g[x])
    print("Visited     :", visited)
    print("Unvisited   :", unvisited)
    print("Path        :", " -> ".join(get_path(x)))

    for y in g[x]:
        if y not in visited and y not in q:
            q.append(y)
            parent[y] = x

    step += 1

# Final Output
print("\nOptimal Solution (Shortest Paths from Source):")
for node in v:
    if node in parent:
        print(f"{start} -> {node} : {' -> '.join(get_path(node))}")
    else:
        print(f"{start} -> {node} : No path")
